            function setActiveLink(links, activeLink) {

                links.forEach(link => {

                    link.classList.remove('border-indigo-500', 'text-neutral-900', 'bg-indigo-50', 'text-indigo-700');

                    link.classList.add('border-transparent', 'text-neutral-500', 'hover:border-neutral-300', 'hover:text-neutral-700');

                    if(link.href === activeLink.href) {

                        link.classList.add('border-indigo-500', 'text-neutral-900');

                        link.classList.remove('border-transparent', 'text-neutral-500');

                    }

                });

            }

            function setActiveMobileLink(links, activeLink) {

                links.forEach(link => {

                    }

                });

            }
